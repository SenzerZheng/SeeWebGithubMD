    int ans = -1;
    int m = strlen(haystack);
    int n = strlen(needle);
    if (n == 0) {
        return 0;
    }
    for (int i = 0; i < m; i++) {
        if (haystack[i] == needle[0]) {
            ans = i;
            for (int j = 0; j < n; j++) {
                if (haystack[i] == needle[j]) {
                    i++;
                    continue;
                } else {
                    ans = -1;
                    break;
                }
            }
        }
    }
    return ans;
